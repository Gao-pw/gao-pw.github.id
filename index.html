2022-05-01

梭哈暴力解法 时间复杂度O(n)

var findRepeatNumber = function(nums) {
    for(let i=0;i<nums.length;i++){
        if(nums.indexOf(nums[i])!== i){
            return nums[i];
        }
    }
};
